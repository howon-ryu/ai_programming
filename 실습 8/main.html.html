def f(x):
    return 4*x**2 + 6*x
def numerical_lim(f,x,h):
    return (f(x+h)-f(x))/h
def Df(x):
    return 8*x+6
xx = np.linspace(-10,10,10)
plt.plot(xx,f(xx),label = 'f(x)')
plt.plot(xx,Df(xx),label='Df(x)')
plt.legend()
plt.show()   #(7)
